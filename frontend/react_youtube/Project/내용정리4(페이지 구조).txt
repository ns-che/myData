2. src/components/section 폴더 생성
컴포넌트 만들기
Header.jsx, Main.jsx Footer.jsx

Header.jsx -> 태그를 header로 <header id='header' role="banner">Header</header>
role은 웹표준 지켜주기 위한 설정 => 역할 = 배너
main.jsx -> 태그를 main으로
Footer.jsx -> 태그를 footer로. 역할 = 사이트의 모든 정보를 보여줌

3. App.js
...
import Header from "./components/Header"
import Main from "./components/Main"
import Footer from "./components/Footer"

const App = () => {
  return (
    <>
      <BrowserRouter>
        <Header /> => Header
        <Main> => Main Start
        <Routes>
          ...
        </Routes>
        </Main> => Main End
        <Footer /> => Footer
      </BrowserRouter>
    </>
  )
}
export default App;
